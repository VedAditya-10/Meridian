# Computer Vision Core
# Principal Note: We MUST use opencv-python-headless. Standard opencv-python
# installs Qt and X11 UI dependencies which bloat the Docker image by ~300MB
# and often crash headless edge servers.
opencv-python-headless>=4.9.0,<4.10.0

# Detection (YOLOv8)
# ultralytics bundles ByteTrack natively via its tracker plugin
ultralytics>=8.1.0

# Linear assignment solver required by ByteTrack
lapx>=0.5.5

# Deep Learning (CPU-only) — the Dockerfile installs via --extra-index-url
# so these will resolve to the CPU wheel from download.pytorch.org/whl/cpu
torch>=2.2.0
onnxruntime>=1.17.0

# numpy < 2.0 required by torch & ultralytics compatibility
numpy<2.0.0

# Telemetry Streaming to Redis
redis>=5.0.1

# Observability
python-json-logger>=2.0.7
